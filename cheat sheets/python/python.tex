%------------ LISTS - SLICING ---------------
\begin{tikzpicture}
\node [mybox] (box){%
	\begin{minipage}{0.3\textwidth}
	\begin{pythoncode}
		a = ['a', 'b', 'c', 'd', 'e']
	\end{pythoncode}

	\vspace*{-0.3cm}
	\includegraphics[width=0.7\linewidth]{img/slicing.pdf}
	
	Syntax \texttt{[start:end]} (start - incl., end - excl., step=1)
	\vspace*{-0.2cm}
	\begin{center}\small{\begin{tabular}{lp{4cm} l}
		\hline
		Explicit start/end \qquad\enspace\enspace\;\; &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[2:4]  # ['c','d']
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
		Implicit end (incl.) &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[2:]   # ['c','d','e']
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
		Implicit start &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[:3]   # ['a','d','e']
		\end{pythoncode}
		\end{minipage}
		\\ \hline

		Negactive indices &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[1:-1] # ['b','c','d']
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
	\end{tabular}}\end{center}

	\vspace*{-0.15cm}	
	Syntax \texttt{[start:end:step]}
	\vspace*{-0.3cm}
	
	\begin{center}\small{\begin{tabular}{lp{4cm} l}
		\hline
		Explicit start/end/step &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[1:5:2]  # ['b','d']
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
		Negative step - backwards &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[4:1:-2] # ['e','d','c']
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
		Implicit start/end/step=1 &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[::] # ['a','b','c','d','e']
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
		No valid index in range &
		\begin{minipage}[t]{\textwidth}
		\begin{pythoncode}
		a[4:2:1]  # []
		\end{pythoncode}
		\end{minipage}
		\\ \hline
		
	\end{tabular}}\end{center}
		
		

	\end{minipage}
